package polyhedra;

 * Polyhedron representing a cylinder.

     * Radius of the cylinder. It must be > 0.

    public Cylinder(double r, double h)
    {
        this.radius = r;
        this.height = h;
    }

        return "Cylinder";

    {
        // null is not a bounding box...

        this.radius = r;
